	PS:指针常量和常量指针
		指针常量（const pointer）:指针本身不能变。
			int a = 10, b = 20;
			int* const p = a;
			
			*p = 100;	// 可以
			p = &b;		// 不可以
		常量指针（pointer to const）：指针指向的内容不能变。
			int a = 10, b = 20;
			const int* p = &a;
			
			*p = 100;   // 不可以
			p = &b;     // 可以

对象的初始化和清理
	构造函数：主要作用在于创建对象时为对象的成员属性赋值，构造函数由编译器自动调用，无须手动调用。
	析构函数：主要作用在于对象销毁前系统自动调用，执行一些清理工作。
	对象的初始化和清理工作是编译器强制要求的，如果不提供构造和析构函数，编译器会提供。编译器提供的构造函数和析构函数是空实现。
	
	类名()	// 构造函数可以有参数，因此可以发送重载
	{
		
	}
	~类名()	// 析构函数不可以有参数，因此不可以发生重载
	{
		
	}
	
	按照参数分类，分为有参构造函数和无参构造函数。无参构造又称为默认构造函数。有参构造函数的三种调用方式：括号法、显示法、隐式转换法。
	按照类型分类，分为普通构造函数和拷贝构造函数
	class person
	{
	public：
		person()
		{
			cout << "无参构造函数" << endl;
		}
		person(int a)
		{
			age = a;
			cout << "有参构造函数" << endl;
		}
		person(const person &p)
		{
			age = p.age;
			cout << "拷贝构造函数" << endl;
		}
		~person()
		{
			cout << "析构函数!" << endl;
		}
	public:
		int age;
	};
	int main(void)
	{
		person p1;				// 调用无参构造函数
		person p2(10);			// 括号法
		person p3 = person(10);	// 显示法
		person p4 = person(p3);	// 显示法
		person p5 = 10;			// 隐式转换法。person p5 = person(10);
		person p6 = p5;			// 隐式转换法。person p6 = person(p5);
	}
	PS:如何显示法和隐式转换法（都是调用的拷贝构造函数）
		person p3 = person(10) 等价于 person(10) + p3 = 临时对象。所以这是用一个对象初始化另一个对象，本质就是调用person(const person &p)。
		person p5 = 10 实际执行的是 person p5 = person(10);
		（只要右边能“变成一个 person 对象”，就可以用来初始化左边）

	C++ 编译器至少给一个类添加 3 个函数：默认构造函数、默认析构函数以及默认拷贝构造函数。
	如果用户定义了有参构造函数，编译器不再提供默认无参构造函数，但是会提供默认拷贝构造函数；如果用户定义了拷贝构造函数，编译器不会再提供其他构造函数。
	
	浅拷贝：简单的赋值拷贝操作。
	深拷贝：在堆区重新申请空间，进行拷贝操作。
	class person
	{
	public:
		person()
		{
			cout << "无参构造函数" << endl;
		}
		person(int age, int height)
		{
			cout << "有参构造函数" << endl;
			
			m_age = age;
			m_height = new int(height);
		}
		person(const person &p)
		{
			cout << "拷贝构造函数" << endl;
			
			m_age = p.m_age;
			m_height = new int(*p.m_height);
		}
		~person()
		{
			cout << "析构函数" << endl;
			
			if(m_height != NULL)
			{
				delete m_height;
			}
		}
	public:
		int m_age;
		int *m_height;
	}

	C++中类的成员可以是另一个类的对象，我们称该对象为成员对象。
	先调用成员对象的构造函数，再调用本类的构造函数。析构顺序与之相反。
	
	静态成员变量
		所有对象共享同一份数据。
		在编译阶段分配内存。
		类内声明，类外初始化。
	静态成员函数
		所有对象共享同一个函数。
		静态成员函数只能访问静态成员变量。

C++对象模型和this指针
	类内的成员变量和成员函数分开存储：函数、静态成员函数、静态成员变量不占对象空间，只有非静态成员变量占对象空间。
	
	非静态成员函数在内存中只有一份实例，多个对象共享该代码，C++ 通过在函数中隐式引入指向当前调用对象的 this 指针来区分具体是哪个对象在调用。
	当形参和成员变量同名时，可用this指针来区分。
	在类的非静态成员函数中返回对象本身，可使用return *this
	class person
	{
	public:
		person(int age)
		{
			this->age = age;
		}
		
		person &personAddperson(person p)
		{
			this->age = age;
			
			return *this;
		}
	public:
		int age;
	}
